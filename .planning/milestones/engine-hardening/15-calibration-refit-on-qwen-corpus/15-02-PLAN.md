```typescript
// Intermediate script — DO NOT commit. Lives at /tmp/15-02-capture-eval.ts.
import { resolve } from "path";
import { register } from "tsconfig-paths";
import { config } from "dotenv";
import { readFileSync, writeFileSync } from "fs";

config({ path: resolve(process.cwd(), ".env.local") });
const tsconfig = JSON.parse(readFileSync(resolve(process.cwd(), "tsconfig.json"), "utf-8"));
register({ baseUrl: process.cwd(), paths: tsconfig.compilerOptions.paths });

// eslint-disable-next-line @typescript-eslint/no-require-imports
const { runEvalOverCorpus } = require("@/lib/engine/corpus/eval-runner");

async function main() {
  const results = await runEvalOverCorpus({
    corpusVersion: "full.2026-05-11",
    maxTotalCostCents: 5000,
    rateLimitDelayMs: 2000,
  });
  // Write a slim row-level capture for Plan 15-03 sweep reuse.
  const capture = results.map((r: any) => ({
    corpus_row_id: r.corpus_row_id,
    actual_bucket: r.actual_bucket,
    predicted_overall_score: r.predicted_overall_score,
    component_scores: r.component_scores ?? null,  // includes platform_fit components
    persona_success_count: r.persona_success_count ?? null,
    niche: r.niche ?? null,
    cost_cents: r.cost_cents ?? 0,
    error: r.error ?? null,
  }));
  writeFileSync("/tmp/15-02-eval-capture.json", JSON.stringify(capture, null, 2));
  console.log(`Captured ${capture.length} rows to /tmp/15-02-eval-capture.json`);
}
main().catch((e) => { console.error(e); process.exit(1); });
```
