```typescript
/**
 * Phase 15 Plan 03 — Threshold + weight sweep on cached Qwen predictions.
 *
 * Reuses /tmp/15-02-eval-capture.json (Plan 15-02 sidecar) to find the
 * (VIRAL_SCORE_CUT, UNDER_SCORE_CUT, platform_fit_weight) triple that
 * maximizes macro_f1 on the full.2026-05-11 corpus under the Qwen pipeline.
 *
 * Bounds per CONTEXT D-09:
 *   viralCut ∈ [55, 80] step 5
 *   underCut ∈ [20, 45] step 5
 *   viralCut > underCut + 20
 * Weight grid per D-11: {0.03, 0.05, 0.07, 0.10}
 *
 * Per Pitfall 3 (15-RESEARCH.md): no engine re-run. Re-bucket from
 * captured predicted_overall_score and re-weight from captured
 * component_scores.
 *
 * Usage:
 *   npx tsx scripts/15-03-sweep-thresholds.ts \
 *     [--capture /tmp/15-02-eval-capture.json] \
 *     [--out /tmp/15-03-sweep-results.json]

import { resolve } from "path";
import { readFileSync, writeFileSync } from "fs";
import { register } from "tsconfig-paths";
import { config } from "dotenv";

config({ path: resolve(__dirname, "../.env.local") });
const tsconfig = JSON.parse(readFileSync(resolve(__dirname, "../tsconfig.json"), "utf-8"));
register({ baseUrl: resolve(__dirname, ".."), paths: tsconfig.compilerOptions.paths });

// eslint-disable-next-line @typescript-eslint/no-require-imports
const { computeMacroF1 } = require("@/lib/engine/corpus/metrics/macro-f1");
// eslint-disable-next-line @typescript-eslint/no-require-imports
const { SCORE_WEIGHTS } = require("@/lib/engine/aggregator");

type CapturedRow = {
  corpus_row_id: string;
  actual_bucket: "viral" | "average" | "under";
  predicted_overall_score: number | null;
  component_scores: Record<string, number | null> | null;
  persona_success_count: number | null;
  niche: string | null;
  cost_cents: number;
  error: string | null;
};

function bucketFor(score: number, viralCut: number, underCut: number): "viral" | "average" | "under" {
  if (score >= viralCut) return "viral";
  if (score < underCut) return "under";
  return "average";
}

// Re-derive predicted_overall_score under a new platform_fit weight.
// Math mirrors aggregator.ts:815-836: raw_score is the bounded weighted sum.
// Sweep approximation: linearly substitute the platform_fit weight in the
// existing weighted-sum equation. If component_scores is missing, fall back
// to the captured predicted_overall_score (no re-weight).
function reweight(row: CapturedRow, weight: number): number | null {
  if (row.predicted_overall_score === null) return null;
  if (!row.component_scores) return row.predicted_overall_score;
  const cs = row.component_scores;
  const platformFit = cs.platform_fit ?? 0;
  const originalWeight = SCORE_WEIGHTS.platform_fit;  // 0.05 baseline
  // Subtract old weighted term, add new weighted term
  const delta = (weight - originalWeight) * platformFit;
  const newScore = row.predicted_overall_score + delta;
  return Math.max(0, Math.min(100, Math.round(newScore)));
}

function gridPoint(rows: CapturedRow[], viralCut: number, underCut: number, weight: number) {
  const pairs: Array<{ predicted_bucket: string; actual_bucket: string }> = [];
  for (const row of rows) {
    if (row.error !== null) continue;
    const score = reweight(row, weight);
    if (score === null) continue;
    pairs.push({
      predicted_bucket: bucketFor(score, viralCut, underCut),
      actual_bucket: row.actual_bucket,
    });
  }
  const f1 = computeMacroF1(pairs);
  return { viralCut, underCut, weight, macro_f1: f1.macro_f1, per_class: f1.per_class, n: pairs.length };
}

function getArg(argv: string[], flag: string, fallback: string): string {
  const i = argv.findIndex((a) => a === flag || a.startsWith(`${flag}=`));
  if (i < 0) return fallback;
  const a = argv[i]!;
  if (a.includes("=")) return a.split("=", 2)[1]!;
  return argv[i + 1] ?? fallback;
}

async function main() {
  const argv = process.argv.slice(2);
  const capturePath = getArg(argv, "--capture", "/tmp/15-02-eval-capture.json");
  const outPath = getArg(argv, "--out", "/tmp/15-03-sweep-results.json");

  const rows: CapturedRow[] = JSON.parse(readFileSync(capturePath, "utf-8"));
  console.log(`Loaded ${rows.length} captured rows from ${capturePath}`);

  const viralCuts = [55, 60, 65, 70, 75, 80];
  const underCuts = [20, 25, 30, 35, 40, 45];
  const weights = [0.03, 0.05, 0.07, 0.10];

  const grid: ReturnType<typeof gridPoint>[] = [];
  for (const vc of viralCuts) {
    for (const uc of underCuts) {
      if (vc <= uc + 20) continue;  // D-09 bound: viralCut > underCut + 20
      for (const w of weights) {
        grid.push(gridPoint(rows, vc, uc, w));
      }
    }
  }
  grid.sort((a, b) => b.macro_f1 - a.macro_f1);
  const best = grid[0]!;

  // Per-niche breakdown at best triple
  const niches = Array.from(new Set(rows.map((r) => r.niche).filter(Boolean))) as string[];
  const perNiche: Record<string, number> = {};
  for (const niche of niches) {
    const sub = rows.filter((r) => r.niche === niche && r.error === null);
    const subPairs = sub.map((r) => ({
      predicted_bucket: bucketFor(reweight(r, best.weight) ?? 0, best.viralCut, best.underCut),
      actual_bucket: r.actual_bucket,
    }));
    perNiche[niche] = computeMacroF1(subPairs).macro_f1;
  }

  const result = {
    best,
    grid_top10: grid.slice(0, 10),
    grid_full: grid,
    per_niche_at_best: perNiche,
    row_count: rows.length,
    skipped: rows.filter((r) => r.error !== null || r.predicted_overall_score === null).length,
  };
  writeFileSync(outPath, JSON.stringify(result, null, 2));

  console.log("Best triple:");
  console.log(`  VIRAL_SCORE_CUT=${best.viralCut}`);
  console.log(`  UNDER_SCORE_CUT=${best.underCut}`);
  console.log(`  platform_fit_weight=${best.weight}`);
  console.log(`  macro_f1=${best.macro_f1.toFixed(4)}`);
  console.log("");
  console.log("Top 10 grid points:");
  for (const g of grid.slice(0, 10)) {
    console.log(`  vc=${g.viralCut} uc=${g.underCut} w=${g.weight} macro_f1=${g.macro_f1.toFixed(4)}`);
  }
  console.log("");
  console.log("Per-niche macro_f1 at best triple:");
  for (const [niche, f1] of Object.entries(perNiche)) {
    console.log(`  ${niche}: ${f1.toFixed(4)}`);
  }
  console.log(`Full grid written to ${outPath}`);
}

main().catch((e) => { console.error(e); process.exit(1); });
```
