```typescript
import { RetentionChart } from '@/components/board/audience/RetentionChart';
import { SegmentTable } from '@/components/board/audience/SegmentTable';
import { CraftFilmstrip } from '@/components/board/content-analysis/CraftFilmstrip';
import { usePermalinkFilmstrips } from '@/hooks/queries/use-permalink-filmstrips';
import {
  toRetentionCurve, normalizeCurve, findBiggestDrop, totalDuration,
  buildSegmentGroups, worstBadGroupKey, cohortDropFrame,
} from '@/components/board/audience/audience-derive';
import { buildCells, audioMixCaption } from '@/components/board/content-analysis/content-analysis-derive';

const filmstrips = usePermalinkFilmstrips();                         // Record<number,string>, store-free (param-free; reads route id internally)
const raw   = data.heatmap?.weighted_curve ?? null;                 // engine emits weighted_curve
const curve = raw ? toRetentionCurve(normalizeCurve(raw)) : null;   // survival curve
const drop  = curve ? findBiggestDrop(normalizeCurve(curve)) : null;
const total = totalDuration(data.heatmap?.segments, 30);            // SECONDS
const nichePct = data.heatmap?.niche_completion_pct != null
  ? (data.heatmap.niche_completion_pct > 1.5 ? data.heatmap.niche_completion_pct/100 : data.heatmap.niche_completion_pct)
  : null;
```

RetentionChartProps: { curve: number[]|null; heatmap: HeatmapPayload|null; drop: BiggestDrop|null; totalDurationSec: number; filmstrips: Record<number,string>; nicheCompletionPct: number|null; isLoading: boolean }
CraftFilmstrip Props: { cells: CraftCell[]; durationSec: number; arc: EmotionArcPoint[]; audio: GeminiAudioSignals|null; audioCaption: string; ctaPresent: boolean; isLoading: boolean }
SegmentTableProps: { groups: SegmentGroup[]; badKey: SlotKey|null; cohortFrames?: Partial<Record<SlotKey,CohortDropFrame>>; isLoading: boolean }

CraftFilmstrip inputs: cells = buildCells(segments, filmstrips, arc, total); arc = data.emotion_arc ?? []; audio = data.audio_signals ?? null; audioCaption = audio ? audioMixCaption(audio) : '' ; ctaPresent from the close mark (see content-analysis-derive).
SegmentTable inputs: groups = buildSegmentGroups(data.heatmap ?? null, data.persona_simulation_results); badKey = worstBadGroupKey(groups); cohortFrames per-slot via cohortDropFrame(data.heatmap ?? null, key, filmstrips).

formatTime TRAP (Pitfall 2): use audience-derive.formatTime (SECONDS) / content-analysis-derive.formatTimeSec — NEVER verdict/TopFixesList.formatTime (ms → the 0:08-vs-0:00 bug). The charts already use the right ones internally; do not pass a ms value.
