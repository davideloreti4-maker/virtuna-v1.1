    ```typescript
    /**
     * Phase 6 audio fingerprint stage (D-A4, D-F0, D-F1, D-F2, D-F3).
     *
     * Embeds the Gemini-emitted audio_description via gemini-embedding-001 (768-dim) and
     * queries the match_trending_sound_by_audio RPC for cosine similarity above threshold.
     *
     * Graceful degradation contract: never throws. On any failure (no description, embedding error,
     * RPC error, no match above threshold), returns null. The aggregator/trends.ts handle the fallback
     * to Jaro-Winkler caption match.
     *
     * Pipeline integration: pipeline.ts wraps this in timed("audio_fingerprint", ...) for stage events;
     * this function does NOT emit its own events.
     */

    import * as Sentry from "@sentry/nextjs";
    import { GoogleGenAI } from "@google/genai";
    import { createLogger } from "@/lib/logger";
    import type { SupabaseClient } from "@supabase/supabase-js";
    import type { AudioFingerprintResult } from "./types";

    const log = createLogger({ module: "audio_fingerprint" });

    // Locked to gemini-embedding-001 per RESEARCH §"State of the Art" (text-embedding-004 shut down 2026-01-14).
    const GEMINI_EMBEDDING_MODEL =
      process.env.GEMINI_EMBEDDING_MODEL ?? "gemini-embedding-001";
    const EMBEDDING_DIMENSIONALITY = 768; // matches migration vector(768)
    // Threshold env-overridable per RESEARCH Open Question #1 (default 0.80).
    const SIMILARITY_THRESHOLD = Number(
      process.env.AUDIO_FINGERPRINT_SIMILARITY_THRESHOLD ?? "0.80",
    );

    let client: GoogleGenAI | null = null;
    function getClient(): GoogleGenAI {
      if (!client) {
        const apiKey = process.env.GEMINI_API_KEY;
        if (!apiKey) {
          throw new Error("Missing GEMINI_API_KEY environment variable");
        }
        client = new GoogleGenAI({ apiKey });
      }
      return client;
    }

    /**
     * Match an audio description against trending_sounds via pgvector cosine similarity.
     *
     * @param audioDescription - 50-150 char description from Gemini's audio_signals.audio_description.
     *   When null/empty, returns null immediately (no Gemini call).
     * @param supabase - Service client (passed in from pipeline.ts).
     * @returns AudioFingerprintResult on match above threshold; null on miss/error.
     *
     * Never throws — all errors are caught + logged + Sentry-reported.
     */
    export async function matchAudioFingerprint(
      audioDescription: string | null | undefined,
      supabase: SupabaseClient,
    ): Promise<AudioFingerprintResult | null> {
      if (!audioDescription || audioDescription.trim().length === 0) {
        log.debug("No audio_description provided — skipping fingerprint match");
        return null;
      }

      try {
        const ai = getClient();
        const response = await ai.models.embedContent({
          model: GEMINI_EMBEDDING_MODEL,
          contents: audioDescription,
          config: {
            outputDimensionality: EMBEDDING_DIMENSIONALITY,
            taskType: "SEMANTIC_SIMILARITY",
          },
        });

        const queryEmbedding = response.embeddings?.[0]?.values;
        if (!queryEmbedding || queryEmbedding.length === 0) {
          log.warn("embedContent returned no embedding values");
          return null;
        }

        const { data: matches, error } = await supabase.rpc(
          "match_trending_sound_by_audio",
          {
            query_embedding: queryEmbedding as unknown as string, // pgvector accepts number[] in JS; types may say string
            match_threshold: SIMILARITY_THRESHOLD,
            match_count: 1,
          },
        );

        if (error) {
          log.warn("pgvector match failed", { error: error.message });
          return null;
        }

        const best = matches?.[0];
        if (!best) {
          log.debug("No match above threshold", { threshold: SIMILARITY_THRESHOLD });
          return null;
        }

        const result: AudioFingerprintResult = {
          sound_name: best.sound_name,
          sound_url: best.sound_url,
          similarity: best.similarity,
          trend_phase: (best.trend_phase as AudioFingerprintResult["trend_phase"]) ?? null,
          velocity_score: Number(best.velocity_score) || 0,
        };

        log.info("Audio fingerprint match", {
          sound_name: result.sound_name,
          similarity: result.similarity,
          trend_phase: result.trend_phase,
        });

        return result;
      } catch (error) {
        Sentry.captureException(error, { tags: { stage: "audio_fingerprint" } });
        const message = error instanceof Error ? error.message : String(error);
        log.warn("Audio fingerprint failed", { error: message });
        return null;
      }
    }
    ```

At runtime, passing a number[] works (Supabase JS handles serialization).     - `Number(best.velocity_score) || 0` — Supabase numeric columns deserialize as string by default; coerce.

    ```typescript
    /**
     * Phase 6 (D-F4 first mechanism) — one-time backfill script.
     *
     * Iterates trending_sounds rows WHERE audio_embedding IS NULL, fetches metadata,
     * generates a Gemini-produced audio_description, embeds via gemini-embedding-001 (768-dim),
     * and upserts audio_embedding + audio_description columns.
     *
     * Idempotency: the WHERE audio_embedding IS NULL clause + ORDER BY id ASC + cursor pagination
     * makes the script resumable on failure (next run skips already-embedded rows).
     *
     * Run: `pnpm tsx scripts/backfill-trending-sound-embeddings.ts`
     * Cost ceiling per D-F4: ~$0.0005/sound × ~50 sounds/day = ~$0.025/day.
     */

    import { config } from "dotenv";
    import { resolve } from "path";
    import { GoogleGenAI } from "@google/genai";
    import { createClient } from "@supabase/supabase-js";
    import type { Database } from "../src/types/database.types";

    config({ path: resolve(__dirname, "../.env.local") });

    const BATCH_SIZE = 25;
    const RATE_LIMIT_MS = 200;
    const GEMINI_EMBEDDING_MODEL = "gemini-embedding-001";
    const EMBEDDING_DIMENSIONALITY = 768;
    const MAX_CONSECUTIVE_FAILURES = 3;

    type TrendingSoundRow = {
      id: string;
      sound_name: string;
      sound_url: string | null;
    };

    /**
     * Generate a 50-150 char audio description from sound metadata.
     * Currently uses sound_name as a synthetic description. Future enhancement (Phase 10+):
     * download sound_url + call Gemini audio analysis for richer description.
     */
    function generateDescriptionFromMetadata(row: TrendingSoundRow): string {
      // Synthetic for v1: pad sound_name to a description-like length.
      // Per CONTEXT D-F4 + RESEARCH §"Pattern": real description comes from feeding sound_url
      // to Gemini for analysis. v1 uses name only as the description; Phase 10 may upgrade.
      const base = `Trending TikTok sound titled "${row.sound_name}"`;
      const url = row.sound_url ? ` (source: ${row.sound_url.split("?")[0]})` : "";
      return `${base}${url}`.slice(0, 280);
    }

    async function embedBatch(
      ai: GoogleGenAI,
      descriptions: string[],
    ): Promise<number[][]> {
      const result = await ai.models.embedContent({
        model: GEMINI_EMBEDDING_MODEL,
        contents: descriptions,
        config: {
          outputDimensionality: EMBEDDING_DIMENSIONALITY,
          taskType: "SEMANTIC_SIMILARITY",
        },
      });
      if (!result.embeddings || result.embeddings.length !== descriptions.length) {
        throw new Error(
          `Expected ${descriptions.length} embeddings, got ${result.embeddings?.length ?? 0}`,
        );
      }
      return result.embeddings.map((e) => {
        if (!e.values) throw new Error("Embedding has no values");
        return e.values;
      });
    }

    export async function main(): Promise<void> {
      const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL;
      const supabaseKey = process.env.SUPABASE_SERVICE_ROLE_KEY;
      if (!supabaseUrl || !supabaseKey) {
        console.error("[backfill] Missing NEXT_PUBLIC_SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY in .env.local");
        process.exit(1);
      }
      const geminiKey = process.env.GEMINI_API_KEY;
      if (!geminiKey) {
        console.error("[backfill] Missing GEMINI_API_KEY in .env.local");
        process.exit(1);
      }

      const supabase = createClient<Database>(supabaseUrl, supabaseKey);
      const ai = new GoogleGenAI({ apiKey: geminiKey });

      let processedCount = 0;
      let consecutiveFailures = 0;
      let cursor: string | null = null;

      while (true) {
        // Idempotent: WHERE audio_embedding IS NULL skips already-processed rows.
        let query = supabase
          .from("trending_sounds")
          .select("id, sound_name, sound_url")
          .is("audio_embedding", null)
          .order("id", { ascending: true })
          .limit(BATCH_SIZE);
        if (cursor) {
          query = query.gt("id", cursor);
        }
        const { data: batch, error: fetchError } = await query;
        if (fetchError) {
          console.error("[backfill] Fetch failed:", fetchError.message);
          process.exit(1);
        }
        if (!batch || batch.length === 0) {
          console.log(`[backfill] Done — processed ${processedCount} sounds total`);
          break;
        }

        const descriptions = batch.map((row) => generateDescriptionFromMetadata(row));
        let embeddings: number[][] = [];
        try {
          embeddings = await embedBatch(ai, descriptions);
          consecutiveFailures = 0;
        } catch (err) {
          consecutiveFailures += 1;
          console.warn(
            `[backfill] Batch embedding failed (consecutive=${consecutiveFailures}):`,
            err instanceof Error ? err.message : String(err),
          );
          if (consecutiveFailures >= MAX_CONSECUTIVE_FAILURES) {
            console.error("[backfill] Aborting after consecutive failures — re-run to resume");
            process.exit(1);
          }
          // Move cursor past this batch so the next iteration tries a different batch.
          cursor = batch[batch.length - 1].id;
          await new Promise((r) => setTimeout(r, 1000)); // backoff before retry
          continue;
        }

        // Update each row with its embedding + description.
        for (let i = 0; i < batch.length; i++) {
          const row = batch[i];
          const { error: updateError } = await supabase
            .from("trending_sounds")
            .update({
              audio_embedding: embeddings[i] as unknown as string,
              audio_description: descriptions[i],
            })
            .eq("id", row.id);
          if (updateError) {
            console.warn(
              `[backfill] Update failed for id=${row.id}:`,
              updateError.message,
            );
            // Row stays with audio_embedding NULL — next run picks it up.
            continue;
          }
          processedCount += 1;
        }

        cursor = batch[batch.length - 1].id;
        console.log(
          `[backfill] Batch done — ${processedCount} processed, cursor=${cursor}`,
        );
        // Rate limit between batches per RESEARCH Pitfall 3.
        await new Promise((r) => setTimeout(r, RATE_LIMIT_MS));
      }
    }

    // Allow being imported in tests without auto-running.
    if (require.main === module) {
      main().catch((err) => {
        console.error("[backfill] Fatal:", err);
        process.exit(1);
      });
    }
    ```
