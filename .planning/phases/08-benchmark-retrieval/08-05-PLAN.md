    ```typescript
    export class EmbedCorpusArgsError extends Error {
      usage: string;
      constructor(message: string, usage: string) {
        super(message);
        this.usage = usage;
      }
    }

    export interface EmbedCorpusArgs {
      mode: "backfill" | "derive-percentiles";
      table?: "training_corpus" | "scraped_videos" | "both";
      batchSize: number;
      dryRun: boolean;
      reEmbedAll: boolean;
    }

    const USAGE = `Usage: pnpm tsx scripts/embed-corpus.ts [options]

Options:
  --backfill                  Embed rows where embedding IS NULL (default mode)
  --derive-percentiles        Compute NON_CORPUS_ENGAGEMENT_PERCENTILES for non-calibrated niches and print code block
  --table=NAME                Limit to 'training_corpus' | 'scraped_videos' | 'both' (default: both)
  --batch-size=N              Texts per Gemini embed call, max 100 (default: 50)
  --dry-run                   Print what would happen without writing
  --re-embed-all              DANGER: UPDATE all embeddings, including non-null. Use after D-06 formula changes only.
  --help                      Show this message

Examples:
  pnpm tsx scripts/embed-corpus.ts
  pnpm tsx scripts/embed-corpus.ts --table=training_corpus --batch-size=100
  pnpm tsx scripts/embed-corpus.ts --derive-percentiles
  pnpm tsx scripts/embed-corpus.ts --re-embed-all --dry-run
`;

    export function parseEmbedCorpusArgs(argv: string[]): EmbedCorpusArgs {
      if (argv.includes("--help") || argv.includes("-h")) {
        throw new EmbedCorpusArgsError("Help requested", USAGE);
      }
      const args: EmbedCorpusArgs = {
        mode: argv.includes("--derive-percentiles") ? "derive-percentiles" : "backfill",
        table: "both",
        batchSize: 50,
        dryRun: argv.includes("--dry-run"),
        reEmbedAll: argv.includes("--re-embed-all"),
      };
      for (const a of argv) {
        if (a.startsWith("--table=")) {
          const v = a.slice("--table=".length);
          if (v !== "training_corpus" && v !== "scraped_videos" && v !== "both") {
            throw new EmbedCorpusArgsError(`--table must be 'training_corpus' | 'scraped_videos' | 'both'; got '${v}'`, USAGE);
          }
          args.table = v;
        }
        if (a.startsWith("--batch-size=")) {
          const n = parseInt(a.slice("--batch-size=".length), 10);
          if (!Number.isInteger(n) || n <= 0 || n > 100) {
            throw new EmbedCorpusArgsError(`--batch-size must be 1..100; got '${a}'`, USAGE);
          }
          args.batchSize = n;
        }
      }
      return args;
    }
    ```

    ```typescript
    const { parseEmbedCorpusArgs, EmbedCorpusArgsError } = require("../src/lib/engine/retrieval/cli/embed-corpus-args");

    const log = (msg: string) => console.log(`[embed-corpus] ${msg}`);
    const warn = (msg: string) => console.warn(`[embed-corpus] WARN: ${msg}`);

    const PAGE = 200;
    const PERCENTILE_MIN_POOL_SIZE = 30; // RESEARCH Finding 5

    type Table = "training_corpus" | "scraped_videos";

    async function backfillTable(supabase: any, table: Table, args: any) {
      let totalEmbedded = 0;
      let totalSkipped = 0;
      let offset = 0;

      while (true) {
        const baseQ = supabase.from(table).select(
          table === "training_corpus"
            ? "id, niche, creator_handle, caption, hashtags"
            : "id, primary_niche, creator_handle, description, hashtags"
        );
        const filterQ = args.reEmbedAll ? baseQ : baseQ.is("embedding", null);
        const { data, error } = await filterQ.order("id").range(offset, offset + PAGE - 1);
        if (error) {
          throw new Error(`Select from ${table} failed: ${error.message}`);
        }
        if (!data || data.length === 0) break;

        // Skip rows with all-null embeddable fields (RESEARCH lines 1170-1174)
        const usable = data.filter((r: any) => {
          const hasCaption = (table === "training_corpus" ? r.caption : r.description);
          const hasHandle = r.creator_handle;
          const hasTags = r.hashtags && r.hashtags.length > 0;
          return hasCaption || hasHandle || hasTags;
        });
        totalSkipped += data.length - usable.length;

        // Embed in batches
        for (let i = 0; i < usable.length; i += args.batchSize) {
          const slice = usable.slice(i, i + args.batchSize);
          const texts = slice.map((r: any) => {
            // For training_corpus: invert alias 'edu' → 'education' for D-06 subject text byte-identity
            // (predict-time uses NICHE_TREE form 'education'; backfill must match — RESEARCH lines 1067-1074)
            const reverseAlias = Object.entries(CORPUS_NICHE_ALIASES).find(([, v]) => v === r.niche)?.[0];
            const slug = table === "training_corpus" ? (reverseAlias ?? r.niche) : r.primary_niche;
            return buildSubjectText({
              primary_slug: slug,
              creator_handle: r.creator_handle ?? null,
              caption: (table === "training_corpus" ? r.caption : r.description) ?? null,
              hashtags: r.hashtags ?? null,
            });
          });

          if (args.dryRun) {
            log(`DRY-RUN would embed batch of ${slice.length} from ${table}`);
            totalEmbedded += slice.length;
            continue;
          }

          try {
            const { vectors } = await embedBatch(texts);
            // Update each row's embedding column
            await Promise.all(slice.map((r: any, j: number) =>
              supabase.from(table).update({ embedding: vectors[j] }).eq("id", r.id)
            ));
            totalEmbedded += slice.length;
          } catch (err: any) {
            warn(`Batch failed (${table}, offset ${offset + i}): ${err.message ?? String(err)} — leaving embedding NULL; re-run will retry`);
            // Soft retry on 429: sleep 60s then continue with next batch (RESEARCH lines 522-523)
            if (String(err.message ?? "").includes("429") || String(err.message ?? "").includes("RESOURCE_EXHAUSTED")) {
              log("Rate limit hit — sleeping 60s");
              await new Promise(r => setTimeout(r, 60_000));
            }
          }
        }

        offset += PAGE;
      }

      log(`${table}: embedded ${totalEmbedded} rows, skipped ${totalSkipped} (all-null fields)`);
    }

    async function derivePercentiles(supabase: any) {
      const NON_CALIBRATED = ["tech-gadgets", "gaming", "fashion-style", "music-performance", "food-cooking"];
    ```
