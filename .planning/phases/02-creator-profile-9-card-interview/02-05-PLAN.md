// src/lib/schemas/creator-profile.ts
export const creatorProfilePatchSchema = z.object({
  target_platforms: z.array(z.enum(["tiktok", "instagram", "youtube"])).max(3).optional(),
  niche_primary: z.string().max(64).nullable().optional(),
  niche_sub: z.string().max(64).nullable().optional(),
  target_audience: z.object({
    age_range: z.enum(["13-17", "18-24", "25-34", "35-44", "45+"]).nullable(),
    gender_skew: z.enum(["female", "balanced", "male"]).nullable(),
    geo: z.string().max(80).nullable(),
    language: z.string().max(80).nullable(),
  }).nullable().optional(),
  primary_goal: z.enum(["growth", "engagement", "brand_deals", "conversion"]).nullable().optional(),
  creator_stage: z.enum(["new", "growing", "established"]).nullable().optional(),
  content_style: z.enum(["talking_head", "b_roll", "educational", "comedy", "tutorial", "vlog"]).nullable().optional(),
  cuts_per_second: z.enum(["slow", "medium", "fast"]).nullable().optional(),
  reference_creators: z.array(z.object({ handle_or_url: z.string().max(256) })).max(3).optional(),
  past_wins: z.array(z.object({ url: z.string().max(512) })).max(2).optional(),
  past_flops: z.array(z.object({ url: z.string().max(512) })).max(2).optional(),
  posting_frequency: z.enum(["daily", "3-4_per_week", "1-2_per_week", "few_per_month", "rarely"]).nullable().optional(),
  time_of_day_aware: z.boolean().nullable().optional(),
  pain_points: z.string().max(500).nullable().optional(),
});
export type CreatorProfilePatch = z.infer<typeof creatorProfilePatchSchema>;

// src/app/api/profile/creator-profile/route.ts
export async function GET(): Promise<Response>; // returns 200 with creator_profiles row or 404
export async function PATCH(request: Request): Promise<Response>; // returns 200 or 400/401/500

// src/hooks/queries/use-creator-profile.ts
export interface CreatorProfileResponse {
  profile_interview_seen_at: string | null;
  target_platforms: string[] | null;
  niche_primary: string | null;
  niche_sub: string | null;
  target_audience: { age_range: string | null; gender_skew: string | null; geo: string | null; language: string | null } | null;
  primary_goal: string | null;
  creator_stage: string | null;
  content_style: string | null;
  cuts_per_second: string | null;
  reference_creators: Array<{ handle_or_url: string }> | null;
  past_wins: Array<{ url: string }> | null;
  past_flops: Array<{ url: string }> | null;
  posting_frequency: string | null;
  time_of_day_aware: boolean | null;
  pain_points: string | null;
}
export function useCreatorProfile(): UseQueryResult<CreatorProfileResponse>;
export function useUpdateCreatorProfile(): UseMutationResult<unknown, Error, Partial<CreatorProfileResponse>>;

queryKeys.profile.creatorProfile = () => ["profile", "creator-profile"] as const;
