    Find and document Raycast's red/pink accent color:
    ```javascript
    (() => {
      // Look for red/pink colored elements
      const allElements = document.querySelectorAll('*');
      const redElements = [];

      allElements.forEach(el => {
        const styles = window.getComputedStyle(el);
        const bgColor = styles.backgroundColor;
        const color = styles.color;
        const borderColor = styles.borderColor;

        // Check for red/pink hues (rough detection)
        [bgColor, color, borderColor].forEach(c => {
          if (c && c.includes('rgb') && !c.includes('rgba(0, 0, 0')) {
            const match = c.match(/rgb\((\d+),\s*(\d+),\s*(\d+)\)/);
            if (match) {
              const [_, r, g, b] = match.map(Number);
              // Red/pink detection: high red, lower green/blue
              if (r > 150 && r > g * 1.5 && r > b * 1.2) {
                redElements.push({ element: el.tagName + '.' + el.className?.split(' ')[0], color: c });
              }
            }
          }
        });
      });

      return { brandColors: [...new Set(redElements.map(e => e.color))].slice(0, 5), samples: redElements.slice(0, 5) };
    })();
    ```

    ```javascript
    (() => {
      // Get @font-face declarations
      const fontFaces = [];
      Array.from(document.styleSheets).forEach(sheet => {
        try {
          Array.from(sheet.cssRules).forEach(rule => {
            if (rule instanceof CSSFontFaceRule) {
              fontFaces.push({
                family: rule.style.fontFamily,
                src: rule.style.src,
                weight: rule.style.fontWeight,
                style: rule.style.fontStyle
              });
            }
          });
        } catch (e) { /* cross-origin sheets */ }
      });

      // Get used font-family values
      const usedFonts = new Set();
      document.querySelectorAll('h1, h2, h3, p, a, button, span').forEach(el => {
        usedFonts.add(window.getComputedStyle(el).fontFamily);
      });

      // Get all font-weight values used
      const usedWeights = new Set();
      document.querySelectorAll('*').forEach(el => {
        const weight = window.getComputedStyle(el).fontWeight;
        if (weight && weight !== 'normal') usedWeights.add(weight);
      });

      return {
        fontFaceDeclarations: fontFaces,
        usedFontFamilies: [...usedFonts],
        usedFontWeights: [...usedWeights].sort()
      };
    })();
    ```
