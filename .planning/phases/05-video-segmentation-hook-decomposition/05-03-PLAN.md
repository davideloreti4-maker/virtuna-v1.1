// =====================================================
// Content-type-aware CTA penalty: when the CTA segment reports cta_present=false on a content
// type that conventionally requires a CTA, deduct points from gemini_score.
//
// Magnitude interpretation (CONTEXT D-06 says "0.5 score units" / "0.3 score units"):
//   We interpret these as 5 and 3 POINTS on the 0-100 gemini_score scale.
//   Rationale: gemini_score = round(avg(factor scores 0-10) * 10) → 0-100 scale. The D-06 deductions are
//   intended as sub-score nudges, not 0.5%-of-100 cosmetic adjustments. Phase 10 ML audit revisits.
//
// Authority: aggregator.ts (this function); inputs sourced from
//   - geminiResult.analysis.cta_segment (Phase 5 — populated by Plan 02 mergeSegments)
//   - pipelineResult.wave0Result.content_type.type (Phase 4 — populated by Wave 0 detector)
// When cta_present=true: NO penalty applied (Claude's Discretion to blend strength deferred).
// When content_type=null OR cta_segment=null: NO penalty (provenance already redistributes).
const CTA_PENALTY_POINTS: Record<string, number> = {
  tutorial: 5,
  b_roll: 3,
  // talking_head, vlog, comedy, slideshow, action, other → no entry → 0 penalty
};

import type { CtaSegmentResult } from "./types";

/**
 * D-06: Penalize gemini_score when CTA expected (tutorial / b_roll) but missing.
 * Pure function; clamps result to [0, 100].
 */
export function applyCtaPenalty(
  geminiScore: number,
  contentTypeSlug: string | null,
  ctaSegment: CtaSegmentResult | null | undefined,
): number {
  // No CTA segment data → no penalty (provenance already redistributes weight elsewhere).
  if (!ctaSegment) return geminiScore;
  // CTA present → no penalty (strength blending deferred per Claude's Discretion).
  if (ctaSegment.cta_present) return geminiScore;
  // Unknown content type → no penalty (Wave 0 failure path).
  if (contentTypeSlug === null) return geminiScore;
  // Lookup penalty per content type; neutral by default.
  const penalty = CTA_PENALTY_POINTS[contentTypeSlug] ?? 0;
  // Clamp to 0-100.
  return Math.max(0, Math.min(100, geminiScore - penalty));
}

// Phase 5 D15 — Rationale-vs-score consistency check (G7 + M9 in AI-SPEC §6/§7)
// =====================================================
// Lightweight regex-based catch for the most common schema-valid-but-semantically-invalid pattern:
// model says "no on-screen text" / "no spoken words" / "silent" in rationale but emits a non-zero
// score for that modality. Phase 5 ships the simple regex; LLM-judge full version is Phase 12 territory.
//
// Emits pipeline_warning with stage: "rationale_inconsistency" — does NOT block (the merged result
// is still usable; the warning fuels M9 production telemetry per AI-SPEC §7).

interface RationaleScoreCheck {
  fieldName: string;
  scoreValue: number;
  rationale: string;
  scoreThreshold: number;     // when score > threshold, expect rationale NOT to contain absent-language
  absentPatterns: RegExp[];
}

const ABSENT_TEXT_PATTERNS: RegExp[] = [
  /\bno (?:on-screen )?text(?: overlay)?\b/i,
  /\bno (?:visible |on-screen )?(?:overlay|caption)s?\b/i,
];
const ABSENT_SPEECH_PATTERNS: RegExp[] = [
  /\bno (?:spoken|verbal) (?:words?|content|opening)\b/i,
  /\bsilent (?:hook|opening|video)\b/i,
  /\bno (?:speech|narration|voiceover|dialogue)\b/i,
];
const TEMPORAL_DRIFT_PATTERNS: RegExp[] = [
  /\blater in the video\b/i,
  /\bafter the (?:hook|5 seconds?|5s)\b/i,
  /\bat the end\b/i,
  /\b(?:0?[6-9]|[1-9]\d+)s\b/i,  // timestamps > 5s in the hook rationale
];

/**
 * Run rationale-vs-score consistency check on a parsed hook segment.
 * Emits up to 2 pipeline_warning events: "rationale_inconsistency" (D15) and "hook_temporal_drift" (D16).
 * Caller is mergeSegments; runs only on hookOk path.
 */
export function validateRationaleConsistency(
  hookAnalysis: { factors: Array<{ name: string; rationale: string }>; hook_decomposition: { text_overlay_score: number; first_words_speech_score: number } },
  onStageEvent: StageEventCallback | undefined,
): void {
  // Collect ALL rationale text from the hook segment for pattern scanning.
  const allRationales = hookAnalysis.factors.map((f) => f.rationale).join(" ");

  // D15: text_overlay_score should be ≤ 2 when rationale claims no on-screen text.
  if (hookAnalysis.hook_decomposition.text_overlay_score > 2 && ABSENT_TEXT_PATTERNS.some((p) => p.test(allRationales))) {
    onStageEvent?.({
      type: "pipeline_warning",
      message: `Hook rationale claims absent text but text_overlay_score=${hookAnalysis.hook_decomposition.text_overlay_score} (>2)`,
      stage: "rationale_inconsistency",
    });
  }
  // D15: first_words_speech_score should be ≤ 2 when rationale claims no speech.
  if (hookAnalysis.hook_decomposition.first_words_speech_score > 2 && ABSENT_SPEECH_PATTERNS.some((p) => p.test(allRationales))) {
    onStageEvent?.({
      type: "pipeline_warning",
      message: `Hook rationale claims absent speech but first_words_speech_score=${hookAnalysis.hook_decomposition.first_words_speech_score} (>2)`,
      stage: "rationale_inconsistency",
    });
  }
  // D16: hook rationale should NOT reference events later than 5s (HAVEN-class temporal hallucination).
  if (TEMPORAL_DRIFT_PATTERNS.some((p) => p.test(allRationales))) {
    onStageEvent?.({
