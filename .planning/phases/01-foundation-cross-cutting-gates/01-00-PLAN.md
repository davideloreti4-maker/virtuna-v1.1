   # Use a tiny inline node script that drives Playwright directly:
   node -e '
   const { chromium } = require("playwright");
   (async () => {
     const browser = await chromium.launch();
     const ctx = await browser.newContext({
       viewport: { width: 1280, height: 900 },
       colorScheme: "dark",
       reducedMotion: "reduce",
     });
     const page = await ctx.newPage();
     await page.goto("http://localhost:3000/app/dashboard", { waitUntil: "networkidle", timeout: 30000 }).catch(() => {});
     await page.waitForTimeout(1500);
     await page.screenshot({ path: "verification/baselines/dashboard-1280.png", fullPage: false });
