```typescript
// verification/scripts/capture-references.spec.ts
// Captures 9 frozen reference snapshots from linear.app + raycast.com into
// verification/reference/ for the visual-fidelity gate.
//
// Run via: pnpm run capture:refs
// Refresh policy (D-07): manual only, on visible Linear/Raycast redesign.
//
// Fallback (RESEARCH §5 line 1298-1299): if Cloudflare or anti-bot blocks the
// capture, take the screenshot manually in Chrome at the target viewport and
// save with the expected filename into verification/reference/.

import { test } from '@playwright/test';
import path from 'path';

const REFERENCE_DIR = path.resolve(__dirname, '../reference');

const TARGETS = [
  { url: 'https://linear.app',          viewport: { width: 1280, height: 900 },  out: 'linear-desktop-1280.png' },
  { url: 'https://linear.app',          viewport: { width: 768,  height: 1024 }, out: 'linear-tablet-768.png' },
  { url: 'https://linear.app',          viewport: { width: 375,  height: 812 },  out: 'linear-mobile-375.png' },
  { url: 'https://linear.app/features', viewport: { width: 1280, height: 900 },  out: 'linear-bento.png' },
  { url: 'https://linear.app/pricing',  viewport: { width: 1280, height: 900 },  out: 'linear-pricing.png' },
  { url: 'https://raycast.com',         viewport: { width: 1280, height: 900 },  out: 'raycast-desktop-1280.png' },
  { url: 'https://raycast.com',         viewport: { width: 768,  height: 1024 }, out: 'raycast-tablet-768.png' },
  { url: 'https://raycast.com',         viewport: { width: 375,  height: 812 },  out: 'raycast-mobile-375.png' },
  { url: 'https://raycast.com',         viewport: { width: 1280, height: 900 },  out: 'raycast-feature-section.png', selector: '[data-section="features"]' },
];

for (const target of TARGETS) {
  test(`Capture ${target.out}`, async ({ browser }) => {
    const context = await browser.newContext({ viewport: target.viewport, colorScheme: 'dark' });
    const page = await context.newPage();
    try {
      await page.goto(target.url, { waitUntil: 'networkidle', timeout: 60000 });
      await page.waitForTimeout(2000); // settle animations + lazy loads
      const outPath = path.join(REFERENCE_DIR, target.out);
      if (target.selector) {
        const element = page.locator(target.selector).first();
        if (await element.count() === 0) {
          console.warn(`[capture-refs] selector not found on ${target.url}: ${target.selector} — falling back to full-page above-fold screenshot for ${target.out}`);
          await page.screenshot({ path: outPath, fullPage: false });
        } else {
          await element.screenshot({ path: outPath });
        }
      } else {
        await page.screenshot({ path: outPath, fullPage: false }); // above-fold only
      }
      console.log(`[capture-refs] captured ${target.out}`);
    } catch (err) {
      const msg = err instanceof Error ? err.message : String(err);
      console.error(`[capture-refs] FAILED ${target.out} (${target.url}): ${msg}`);
      console.error(`[capture-refs] fallback: open ${target.url} in Chrome at ${target.viewport.width}x${target.viewport.height}, screenshot above-fold, save to verification/reference/${target.out}`);
      // Do NOT throw — let other captures continue. The fallback is manual.
    } finally {
      await context.close();
    }
  });
}
```

Key constraints (per RESEARCH §5 + PATTERNS):

- The 9 TARGETS array matches D-06 exactly (count, URLs, viewports, filenames).
- Per-target viewport is set via `browser.newContext({ viewport })` — the Playwright config's project viewport is overridden so the spec runs once per target regardless of the `--project=desktop` invocation.

- Errors are CAUGHT and LOGGED — don't throw, so a Cloudflare block on linear.app doesn't prevent raycast capture from running.
- `colorScheme: 'dark'` matches Virtuna's dark-first branding.
