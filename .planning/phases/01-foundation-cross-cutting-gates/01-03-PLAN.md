// src/components/landing/landing-motion-provider.tsx
"use client";

import { LazyMotion, domAnimation } from "motion/react";

/**
 * Wraps landing routes with LazyMotion + domAnimation features.
 * Reduces initial motion library bundle from ~34 KB gzipped to ~4.6 KB.
 *
 * `strict` enabled: throws at runtime if any landing component imports `motion.*`
 * instead of `m.*` — this is the linting mechanism we use because we can't
 * easily write an ESLint rule that distinguishes landing-route from non-landing
 * imports.
 *
 * `domAnimation` features (+15 KB async): animations, variants, exit animations,
 * tap/hover/focus gestures.
 * If layout animations or drag/pan are needed in a future phase, swap to
 * `domMax` (+25 KB async).
 */
export function LandingMotionProvider({ children }: { children: React.ReactNode }) {
  return (
    <LazyMotion features={domAnimation} strict>
      {children}
    </LazyMotion>
  );
}
```

Key constraints:

- The `strict` prop is mandatory — it is the runtime lint.
- Component name `LandingMotionProvider` (exact case).

// src/components/MotionWrapper.tsx
"use client";

import * as m from "motion/react-m";
import type { ReactNode, ComponentProps } from "react";

/**
 * Thin "use client" wrapper around animated landing leaves.
 *
 * Pattern (per FOUND-10, RESEARCH §4):
 * - Landing sections are Server Components (NO "use client" — pure markup, SEO-visible).
 * - Animated leaves inside sections are client islands using `m.*` from `motion/react-m`.
 * - NEVER wrap an entire section in <MotionWrapper> — that violates AS-15 (anti-slop
 *   blacklist: section-level Framer Motion wrappers). Section structure stays static.
 *
 * LCP exception: the hero <h1> MUST animate via CSS keyframes (in landing.css), NOT
 * via this wrapper. JS-driven `initial={{ opacity: 0 }}` keeps the element invisible
 * until React hydrates, killing Lighthouse LCP. See RESEARCH §4 lines 517-538.
 *
 * Strict-mode safety: this file consumes `m.*` from `motion/react-m`, which is the
 * import path required under `LazyMotion strict` (LandingMotionProvider). If you see
 * a runtime "You're trying to use motion within a LazyMotion component" error, an
 * upstream caller is using `motion/react` (full bundle) — replace with `motion/react-m`.
 */
export function MotionWrapper({
  children,
  ...motionProps
}: { children: ReactNode } & ComponentProps<typeof m.div>) {
  return <m.div {...motionProps}>{children}</m.div>;
}
```

Key constraints:
- File path is `src/components/MotionWrapper.tsx` (top-level `components/` directory — NOT `components/landing/`). This is the example reference; per PATTERNS line 191-194 it lives at this path so it can be cited from CRAFT-RUBRIC.md.
- Import path is exactly `motion/react-m` (NOT `motion/react`).
- Type the wrapper props as `{ children: ReactNode } & ComponentProps<typeof m.div>` so motion props (`initial`, `animate`, `transition`, `whileInView`, etc.) pass through.
- The `"use client"` directive is present.
