```typescript
      if (pageConfig.referenceBaselines) {
        for (const refBaseline of pageConfig.referenceBaselines) {
          try {
            const element = page.locator(refBaseline.selector).first();
            if (await element.count() === 0) {
              console.log(`  [reference] selector not found: ${refBaseline.selector} — skipping`);
              continue;
            }
            const currentPath = path.join(VIRTUNA_SCREENSHOTS_DIR, `_ref-current-${refBaseline.diffName}`);
            await element.screenshot({ path: currentPath });
            const referencePath = path.resolve(refBaseline.referenceFile);
            const fs = await import('node:fs');
            if (!fs.existsSync(referencePath)) {
              console.log(`  [reference] reference file missing: ${refBaseline.referenceFile} — skipping (Plan 01-12 / 01-00 may not have captured yet)`);
              continue;
            }
            const diffPath = path.join(DIFF_SCREENSHOTS_DIR, refBaseline.diffName);
            generateDiff(currentPath, referencePath, diffPath);
            console.log(`  [reference] ${refBaseline.diffName} written`);
          } catch (err) {
            console.log(`  [reference] error on ${refBaseline.diffName}: ${err instanceof Error ? err.message : String(err)}`);
          }
        }
      }
```
