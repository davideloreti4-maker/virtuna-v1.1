  <action>
Create a premium CTA button with Framer Motion spring physics:

**Implementation:**
```tsx
"use client";

import { motion } from "motion/react";
import Link from "next/link";
import { cn } from "@/lib/utils";

export interface HeroButtonProps {
  children: React.ReactNode;
  href: string;
  className?: string;
}

export function HeroButton({ children, href, className }: HeroButtonProps) {
  return (
    <motion.div
      whileHover={{ scale: 1.05 }}
      whileTap={{ scale: 0.98 }}
      transition={{
        type: "spring",
        stiffness: 400,
        damping: 17,
      }}
    >
      <Link
        href={href}
        className={cn(
          // Base styles
          "inline-flex items-center justify-center",
          "px-8 py-3.5",
          "text-base font-medium",
          "rounded-xl",
          // Colors - solid button per CONTEXT.md
          "bg-white text-black",
          "hover:bg-white/90",
          // Transition for non-spring properties
          "transition-colors duration-200",
          className
        )}
      >
        {children}
      </Link>
    </motion.div>
  );
}
```

**Notes:**
- Uses motion/react (the new import path for framer-motion v12+)
- Spring physics: stiffness 400, damping 17 (per roadmap success criteria)
- Solid white button with black text (premium, high contrast)
- Scale 1.05 on hover, 0.98 on tap for satisfying feedback
- Links to /signup per CONTEXT.md
