```typescript
  targetPlatforms: ReadonlyArray<PlatformTarget>,
```

```typescript
import * as Sentry from "@sentry/nextjs";
import OpenAI from "openai";
import { createLogger } from "@/lib/logger";
import { emitStageStart, emitStageEnd, type StageEventCallback } from "../events";
import { isCircuitOpen } from "../deepseek";
import {
  STABLE_PLATFORM_FIT_SYSTEM_PROMPT,
  buildPlatformFitUserMessage,
  PlatformFitResponseSchema,
  type PlatformTarget,
} from "./platform-fit-prompts";
import type { PlatformFitResult } from "../types";

const logger = createLogger("platform-fit");

const MODEL = process.env.DEEPSEEK_PLATFORM_FIT_MODEL ?? "deepseek-v4-flash";
const CACHE_HIT_PRICE  = 0.0028 / 1_000_000;
const CACHE_MISS_PRICE = 0.14   / 1_000_000;
const OUTPUT_PRICE     = 0.28   / 1_000_000;
const PER_CALL_TIMEOUT_MS = 15_000;

let _client: OpenAI | null = null;
function getClient(): OpenAI {
  if (!_client) {
    const apiKey = process.env.DEEPSEEK_API_KEY;
    if (!apiKey) throw new Error("Missing DEEPSEEK_API_KEY environment variable");
    _client = new OpenAI({ apiKey, baseURL: "https://api.deepseek.com" });
  }
  return _client;
}

// Exported for pipeline.ts integration (Plan 09-07).
// Accepts the same pipeline result shape that aggregateScores() receives.
export async function runPlatformFit(
  // Pass individual fields rather than a typed PipelineResult to avoid circular imports.
  // Plan 09-07 will destructure and pass these from pipeline.ts local variables.
  params: {
    creatorContext: import("../creator").CreatorContext;
    geminiResult: import("../gemini").GeminiAnalysisResult | null;
    featureVector: import("../types").FeatureVector | null;
    personaBehavioralAggregate: import("../wave3/persona-registry").PersonaBehavioralAggregate | null;
  },
  onEvent?: StageEventCallback,
): Promise<PlatformFitResult[] | null> {
  const stageStart = emitStageStart(onEvent, "platform_fit", "post");
  let costCents = 0;

  if (isCircuitOpen()) {
    emitStageEnd(onEvent, "platform_fit", "post", stageStart, {
      cost_cents: 0, ok: false, warning: "circuit_breaker_open",
    });
    return null;
  }

  // Determine target platforms from Card 0 — default to TikTok if empty (D-03)
  const rawPlatforms = params.creatorContext.target_platforms as PlatformTarget[] | undefined;
  const targetPlatforms: ReadonlyArray<PlatformTarget> =
    rawPlatforms && rawPlatforms.length > 0
      ? rawPlatforms.filter((p): p is PlatformTarget =>
          ["tiktok", "ig_reels", "yt_shorts"].includes(p),
        )
      : (["tiktok"] as const);

  const ai = getClient();
  let attempt = 0;
  let lastError: Error | null = null;

  while (attempt <= 1) {
    const controller = new AbortController();
    const timer = setTimeout(() => controller.abort(), PER_CALL_TIMEOUT_MS);
    try {
      const userMessage = buildPlatformFitUserMessage(params, targetPlatforms, attempt > 0);
      const response = await ai.chat.completions.create(
        {
          model: MODEL,
          messages: [
            { role: "system", content: STABLE_PLATFORM_FIT_SYSTEM_PROMPT },
            { role: "user",   content: userMessage },
          ],
          response_format: { type: "json_object" },
        },
        { signal: controller.signal },
      );
      clearTimeout(timer);

      // Cache-aware cost telemetry (verbatim from wave3.ts:180-197)
      const usage = response.usage as unknown as {
        prompt_tokens?: number;
        prompt_cache_hit_tokens?: number;
        prompt_cache_miss_tokens?: number;
        completion_tokens?: number;
      };
      const cacheHit  = usage?.prompt_cache_hit_tokens  ?? 0;
      const cacheMiss = usage?.prompt_cache_miss_tokens ?? 0;
      const completion = usage?.completion_tokens ?? 0;
      const hasBreakdown = cacheHit > 0 || cacheMiss > 0;
      const inputCost = hasBreakdown
        ? cacheHit * CACHE_HIT_PRICE + cacheMiss * CACHE_MISS_PRICE
        : (usage?.prompt_tokens ?? 0) * CACHE_MISS_PRICE;
      costCents += (inputCost + completion * OUTPUT_PRICE) * 100;

      const text = response.choices[0]?.message?.content ?? "{}";
      const parsed = PlatformFitResponseSchema.safeParse(JSON.parse(text));
      if (!parsed.success) {
        lastError = new Error(`validation failed: ${parsed.error.message}`);
        logger.warn("platform_fit Zod validation failed", { attempt, error: lastError.message });
        if (attempt === 0) { attempt++; continue; }
        throw lastError;
      }

      // Pitfall 5: filter to requested platforms only
      const results = parsed.data.platforms.filter(
        (p) => (targetPlatforms as readonly string[]).includes(p.platform),
      );

      emitStageEnd(onEvent, "platform_fit", "post", stageStart, {
        cost_cents: +costCents.toFixed(4), ok: true,
      });
      logger.debug("platform_fit complete", { results: results.length, cost_cents: costCents });
      return results;
    } catch (err) {
      clearTimeout(timer);
      lastError = err instanceof Error ? err : new Error(String(err));
      if (lastError.name === "AbortError" || attempt >= 1) break;
      attempt++;
    }
  }

  Sentry.captureException(lastError, { tags: { stage: "platform_fit" } });
  emitStageEnd(onEvent, "platform_fit", "post", stageStart, {
    cost_cents: +costCents.toFixed(4), ok: false, warning: lastError?.message,
  });
  return null;
}
```
